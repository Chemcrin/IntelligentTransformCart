# 智能物流小车 传感器桥接层文档 V2.0

> 文 档 名：智能物流小车-传感器桥接层文档  
> 文档版本：V2.0  
> 最后更新：2026-05-26  
> 作    用：在 GPT 版 Board_App 之上，给运动控制层提供 rad/m 单位的传感器 API

---

## 1. 设计背景

### 1.1 GPT 版已有的能力

GPT 版代码的 `board_app.h` 已经提供了：

```c
// ✅ 全局快照（deg + mm 单位）
extern volatile SENSOR_Snapshot_t g_sensor_snapshot;

// ✅ 快照获取接口
void Board_App_GetSnapshot(SENSOR_Snapshot_t *out);

// ✅ 自动更新
Board_App_Loop()  →  IMU_App_Loop() + Laser_App_Loop() → board_app_update_snapshot()
```

**桥接层不做重复工作**，而是在 Board_App 之上加一层**单位转换**。

### 1.2 为什么还需要桥接层

```
Board_App（GPT 版）           桥接层                   运动控制层
─────────────────────    ──────────────────    ──────────────────
SENSOR_Snapshot_t        SensorSnapshot_t      ChassisVel_t
  yaw: deg                 yaw_rad: rad          omega: rad/s
  distance: mm             distance_m: m         vx/vy: m/s
  gyro: deg/s              gyro_z_rads: rad/s    PD 控制需要 rad

  Board_App_GetSnapshot()  Sensor_GetSnapshot()  Chassis_ControlTask()
```

桥接层做的事：**读快照 → 做单位转换 → 给运动控制层用**。

## 3. 架构分层图

```text
    运动控制层                   桥接层                     GPT Board_App
    ┌──────────────────┐    ┌────────────────┐    ┌─────────────────────────┐
    │ Chassis_Control  │    │                │    │ g_sensor_snapshot       │
    │   Task()         │───▶│ Sensor_Get     │───▶│  .car_yaw_est_deg  deg  │
    │                  │    │   Snapshot()   │    │  .gyro_yaw_dps     dps  │
    │ SensorSnapshot_t │    │                │    │  .laser_filtered_  mm   │
    │   .yaw_rad       │◀───│ deg→rad        │◀───│   distance_mm           │
    │   .gyro_z_rads   │    │ dps→rad/s      │    │  .laser_is_warning 0/1  │
    │   .distance_m    │    │ mm→m           │    │  .imu_valid        0/1  │
    │   .laser_warning │    │                │    └─────────────────────────┘
    │   .imu_valid     │    │                │            ↑
    └──────────────────┘    └────────────────┘    Board_App_Loop()
```

**要点**：桥接层不读 `g_imu_*` / `g_laser_*` 全局变量，而是读 `g_sensor_snapshot`（已经由 `Board_App_Loop()` 维护好）。

## 4. 头文件 `sensor_bridge.h`

```c
/**
 * @file    sensor_bridge.h
 * @brief   传感器桥接层 —— 给运动控制层提供 rad/m 单位的传感器 API
 * @note    内部依赖 board_app.h 的 SENSOR_Snapshot_t 和 Board_App_GetSnapshot()
 *          运动控制层不要直接读 g_sensor_snapshot，统一走本桥接层
 */

#ifndef SENSOR_BRIDGE_H
#define SENSOR_BRIDGE_H

#include <stdint.h>
#include <stdbool.h>

#ifdef __cplusplus
extern "C" {
#endif

/* ========== IMU 接口 ========== */

/**
 * @brief  获取当前偏航角（弧度制）
 * @return yaw 值，单位 rad，范围 [-pi, pi]
 * @note   从 g_sensor_snapshot.car_yaw_est_deg 转换而来
 */
float JY61P_GetYawRad(void);

/**
 * @brief  获取当前偏航角速度（弧度制）
 * @return gyro_z 值，单位 rad/s
 * @note   从 g_sensor_snapshot.gyro_yaw_dps 转换而来
 */
float JY61P_GetGyroZRadS(void);

/**
 * @brief  获取俯仰角（度，用于姿态异常检测）
 * @return pitch 值，单位 deg
 */
float JY61P_GetPitchDeg(void);

/**
 * @brief  获取横滚角（度，用于姿态异常检测）
 * @return roll 值，单位 deg
 */
float JY61P_GetRollDeg(void);

/**
 * @brief  IMU 数据是否有效
 * @return true = 数据有效，false = 数据不可用
 */
bool JY61P_IsDataValid(void);

/**
 * @brief  IMU 是否已完成校准
 * @return true = 已校准，false = 未校准
 */
bool JY61P_IsCalibrated(void);


/* ========== 激光测距接口 ========== */

/**
 * @brief  获取滤波后距离
 * @param[out] distance_m  输出距离值，单位 m
 * @return true  = 有有效数据，*distance_m 被更新
 *         false = 数据无效/超时，*distance_m 不变
 * @note   从 g_sensor_snapshot.laser_filtered_distance_mm 转换而来
 */
bool Laser_GetDistance(float *distance_m);

/**
 * @brief  获取原始距离（未滤波，单位 mm，供诊断用）
 * @return 原始距离值 mm，若无效返回 0
 */
uint16_t Laser_GetRawDistanceMm(void);

/**
 * @brief  检查激光预警等级
 * @return 0 = 正常，1 = 预警，2 = 减速，3 = 停车
 * @note   内部从 g_sensor_snapshot 的三个标志位组合而来
 */
uint8_t Laser_GetWarningLevel(void);

/**
 * @brief  激光模块是否在线
 * @return true = 模块正常，false = 通信异常
 */
bool Laser_IsOnline(void);


/* ========== 传感器快照 ========== */

/**
 * @brief  传感器数据快照结构体（rad/m 单位版本）
 * @note   在控制周期开始时调用 Sensor_GetSnapshot() 一次获取，
 *          后续运动控制直接读快照字段，不再调用单个函数
 */
typedef struct {
    /* IMU */
    float    yaw_rad;           /* 偏航角，rad       ← g_sensor_snapshot.car_yaw_est_deg */
    float    gyro_z_rads;       /* 偏航角速度，rad/s ← g_sensor_snapshot.gyro_yaw_dps */
    float    pitch_deg;         /* 俯仰角，deg       ← g_sensor_snapshot.car_pitch_deg */
    float    roll_deg;          /* 横滚角，deg       ← g_sensor_snapshot.car_roll_deg */
    bool     imu_valid;         /* IMU 有效          ← g_sensor_snapshot.imu_valid */
    bool     imu_calibrated;    /* IMU 已校准        ← g_sensor_snapshot.imu_calibrated */

    /* 激光 */
    float    distance_m;        /* 激光距离，m       ← g_sensor_snapshot.laser_filtered_distance_mm */
    bool     laser_valid;       /* 激光数据有效      ← g_sensor_snapshot.laser_ok */
    uint8_t  laser_warning;     /* 预警等级 0-3      ← 三标志组合 */

    /* 系统 */
    uint32_t timestamp_ms;      /* 快照时间戳        ← g_sensor_snapshot.timestamp_ms */
    uint32_t fault_flags;       /* 传感器故障标志    ← g_sensor_snapshot.sensor_fault_flags */
} SensorSnapshot_t;

/**
 * @brief  获取传感器数据快照（rad/m 单位）
 * @param[out] snap  输出快照指针，不能为 NULL
 * @note   内部调用 Board_App_GetSnapshot() 得到 deg/mm 数据，再转换单位
 *         在 chassis_control.c 的每个控制周期开始时调用一次
 */
void Sensor_GetSnapshot(SensorSnapshot_t *snap);

#ifdef __cplusplus
}
#endif

#endif /* SENSOR_BRIDGE_H */
```

## 5. 实现文件 `sensor_bridge.c`

```c
/**
 * @file    sensor_bridge.c
 * @brief   传感器桥接层实现
 * @note    依赖 board_app.h，通过 Board_App_GetSnapshot() 获取原始数据
 */

#include "sensor_bridge.h"
#include "board_app.h"   /* 提供 g_sensor_snapshot 和 Board_App_GetSnapshot() */

/* 度 → 弧度换算系数 */
#define DEG_TO_RAD   0.01745329252f

/**
 * @brief   将 deg/mm 单位的 SENSOR_Snapshot_t 转为 rad/m 单位的 SensorSnapshot_t
 */
static void convert_snapshot(const SENSOR_Snapshot_t *in, SensorSnapshot_t *out)
{
    if (in == NULL || out == NULL) return;

    /* IMU */
    out->yaw_rad      = in->car_yaw_est_deg * DEG_TO_RAD;
    out->gyro_z_rads  = in->gyro_yaw_dps * DEG_TO_RAD;
    out->pitch_deg    = in->car_pitch_deg;
    out->roll_deg     = in->car_roll_deg;
    out->imu_valid    = (in->imu_valid != 0U);
    out->imu_calibrated = (in->imu_calibrated != 0U);

    /* 激光 */
    if (in->laser_ok) {
        out->distance_m = (float)in->laser_filtered_distance_mm / 1000.0f;
    } else {
        out->distance_m = -1.0f;   /* 负值表示无效 */
    }
    out->laser_valid   = (in->laser_ok != 0U);
    out->laser_warning = 0U;
    if (in->laser_is_warning)   out->laser_warning = 1U;
    if (in->laser_is_slow)      out->laser_warning = 2U;
    if (in->laser_is_too_close) out->laser_warning = 3U;

    /* 系统 */
    out->timestamp_ms = in->timestamp_ms;
    out->fault_flags  = in->sensor_fault_flags;
}


/* ========== IMU 接口 ========== */

float JY61P_GetYawRad(void)
{
    return (float)g_sensor_snapshot.car_yaw_est_deg * DEG_TO_RAD;
}

float JY61P_GetGyroZRadS(void)
{
    return (float)g_sensor_snapshot.gyro_yaw_dps * DEG_TO_RAD;
}

float JY61P_GetPitchDeg(void)
{
    return g_sensor_snapshot.car_pitch_deg;
}

float JY61P_GetRollDeg(void)
{
    return g_sensor_snapshot.car_roll_deg;
}

bool JY61P_IsDataValid(void)
{
    return (g_sensor_snapshot.imu_valid != 0U);
}

bool JY61P_IsCalibrated(void)
{
    return (g_sensor_snapshot.imu_calibrated != 0U);
}


/* ========== 激光测距接口 ========== */

bool Laser_GetDistance(float *distance_m)
{
    if (distance_m == NULL) return false;
    if (g_sensor_snapshot.laser_ok == 0U) return false;

    *distance_m = (float)g_sensor_snapshot.laser_filtered_distance_mm / 1000.0f;
    return true;
}

uint16_t Laser_GetRawDistanceMm(void)
{
    return g_sensor_snapshot.laser_raw_distance_mm;
}

uint8_t Laser_GetWarningLevel(void)
{
    uint8_t level = 0U;
    if (g_sensor_snapshot.laser_is_warning)   level = 1U;
    if (g_sensor_snapshot.laser_is_slow)      level = 2U;
    if (g_sensor_snapshot.laser_is_too_close) level = 3U;
    return level;
}

bool Laser_IsOnline(void)
{
    return (g_sensor_snapshot.laser_ok != 0U);
}


/* ========== 快照接口 ========== */

void Sensor_GetSnapshot(SensorSnapshot_t *snap)
{
    if (snap == NULL) return;

    /* 从 Board_App 获取原始快照（deg/mm），再做单位转换 */
    SENSOR_Snapshot_t raw;
    Board_App_GetSnapshot(&raw);
    convert_snapshot(&raw, snap);
}
```

## 6. 在 main.c 中集成

```c
#include "board_app.h"
#include "sensor_bridge.h"

int main(void)
{
    /* 初始化 HAL 和外设 */
    HAL_Init();
    SystemClock_Config();
    MX_GPIO_Init();
    MX_I2C1_Init();
    MX_USART1_UART_Init();

    Board_App_Init();                /* 初始化 IMU + 激光 */

    while (1)
    {
        Board_App_Loop();            /* 更新传感器数据到 g_sensor_snapshot */

        SensorSnapshot_t snap;
        Sensor_GetSnapshot(&snap);   /* 获取 rad/m 单位的快照 */

        /* 运动控制（后续加） */
        // Chassis_ControlTask(&snap);

        HAL_Delay(10);
    }
}
```

## 8. 与 V1.0 的主要差异

| 差异点 | V1.0（旧） | V2.0（新） |
|---|---|---|
| 数据来源 | 直读 `g_imu_*` / `g_laser_*` 全局变量 | 通过 `Board_App_GetSnapshot()` 读 `g_sensor_snapshot` |
| 快照缓存 | 自维护 `s_snap_cache` + `s_snap_valid` | 不维护缓存，每次调用实时转换 |
| 架构层次 | 跳过 Board_App，直接对接传感器层 | 严格分层：Board_App → bridge → 运动控制 |
| 快照结构体 | `SensorSnapshot_t`（自定义） | 同名，但新增 `timestamp_ms` 和 `fault_flags` |
| 故障信息 | 无 | 透传 `sensor_fault_flags`，供安全逻辑使用 |
