# 视觉识别改进版（OpenCV）

## 改进点
1. 高斯模糊预处理降噪
2. 用**圆形度**（4π·面积/周长²）代替宽高比，更准确区分球和桶
3. 开运算+闭运算双重形态学处理
4. 可调参数统一放在文件顶部，方便现场标定

## 代码

```python
import cv2
import numpy as np
import os
os.environ['QT_QPA_PLATFORM'] = 'xcb'

# ===== 可调参数区 =====
# HSV阈值（根据实际光照标定）
RED_LOWER1 = np.array([0, 120, 80])
RED_UPPER1 = np.array([10, 255, 255])
RED_LOWER2 = np.array([170, 120, 80])
RED_UPPER2 = np.array([180, 255, 255])
BLUE_LOWER = np.array([100, 120, 80])
BLUE_UPPER = np.array([130, 255, 255])

# 面积阈值
AREA_MIN = 400
AREA_MAX = 50000

# 圆形度阈值（1最圆，越小越扁）
CIRCULARITY_BALL = 0.7   # 球：接近圆形
CIRCULARITY_BUCKET = 0.4 # 桶：偏矩形

# 形态学核大小
KERNEL_SIZE = 5
# ====================

kernel = np.ones((KERNEL_SIZE, KERNEL_SIZE), np.uint8)
cap = cv2.VideoCapture(0)
cap.set(3, 640)
cap.set(4, 480)

while True:
    ret, frame = cap.read()
    if not ret:
        continue

    # 1. 高斯模糊降噪
    blur = cv2.GaussianBlur(frame, (5, 5), 0)

    # 2. 转HSV
    hsv = cv2.cvtColor(blur, cv2.COLOR_BGR2HSV)

    # 3. 颜色阈值
    mask_r1 = cv2.inRange(hsv, RED_LOWER1, RED_UPPER1)
    mask_r2 = cv2.inRange(hsv, RED_LOWER2, RED_UPPER2)
    red_mask = cv2.bitwise_or(mask_r1, mask_r2)
    blue_mask = cv2.inRange(hsv, BLUE_LOWER, BLUE_UPPER)

    # 4. 形态学去噪
    red_mask = cv2.morphologyEx(red_mask, cv2.MORPH_OPEN, kernel)
    red_mask = cv2.morphologyEx(red_mask, cv2.MORPH_CLOSE, kernel)
    blue_mask = cv2.morphologyEx(blue_mask, cv2.MORPH_OPEN, kernel)
    blue_mask = cv2.morphologyEx(blue_mask, cv2.MORPH_CLOSE, kernel)

    def detect_objects(mask, color_label, box_color):
        contours, _ = cv2.findContours(mask, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE)
        for cnt in contours:
            area = cv2.contourArea(cnt)
            if area < AREA_MIN or area > AREA_MAX:
                continue

            # 计算圆形度
            perimeter = cv2.arcLength(cnt, True)
            if perimeter == 0:
                continue
            circularity = 4 * np.pi * area / (perimeter * perimeter)

            x, y, w, h = cv2.boundingRect(cnt)
            cv2.rectangle(frame, (x, y), (x + w, y + h), box_color, 2)

            # 用圆形度区分球和桶
            if circularity > CIRCULARITY_BALL:
                obj_type = "Ball"
            elif circularity > CIRCULARITY_BUCKET:
                obj_type = "Bucket"
            else:
                obj_type = "Other"

            label = f"{color_label} {obj_type}"
            cv2.putText(frame, label, (x, y - 8),
                        cv2.FONT_HERSHEY_SIMPLEX, 0.6, box_color, 2)

    detect_objects(red_mask, "Red", (0, 0, 255))
    detect_objects(blue_mask, "Blue", (255, 0, 0))

    cv2.imshow("Detection", frame)
    if cv2.waitKey(1) & 0xFF == ord('q'):
        break

cap.release()
cv2.destroyAllWindows()
```

## 调参建议
- HSV阈值：在现场光照下拍一张照片，用 `cv2.imshow()` 分别显示红/蓝的掩膜，调阈值直到目标刚好被完整提取、背景不误检
- 圆形度阈值：球≈0.85-0.95，桶≈0.5-0.7，可以先设0.7和0.4再微调
- 如果光线变化大，可以加自适应光照补偿（直方图均衡化）
